--tRUCK pENDING
SELECT     dbo.GATEINDETAIL.GIDID, dbo.GATEINDETAIL.AVHLNO
FROM         dbo.GATEINDETAIL LEFT OUTER JOIN
                      dbo.VW_SHIPPINGBILL_GATEINT_TRUCKNO_CBX_ASSGN ON 
                      dbo.GATEINDETAIL.GIDID = dbo.VW_SHIPPINGBILL_GATEINT_TRUCKNO_CBX_ASSGN.GIDID
WHERE     (dbo.GATEINDETAIL.GPSTYPE IN (1, 5)) AND (dbo.GATEINDETAIL.SDPTID = 2) AND 
                      (dbo.GATEINDETAIL.GPNOP - ISNULL(dbo.VW_SHIPPINGBILL_GATEINT_TRUCKNO_CBX_ASSGN.SBDNOP, 0) > 0)
and (GIDATE >='01-APR-2015')                      
ORDER BY dbo.GATEINDETAIL.AVHLNO

--cONTAINER pENDING
SELECT     dbo.GATEINDETAIL.GIDID, dbo.GATEINDETAIL.CONTNRNO
FROM         dbo.GATEINDETAIL LEFT OUTER JOIN
                      dbo.STUFFINGDETAIL ON dbo.GATEINDETAIL.GIDID = dbo.STUFFINGDETAIL.GIDID
WHERE (dbo.GATEINDETAIL.DispStatus = 0) and      (dbo.GATEINDETAIL.GPSTYPE = 4)  AND (ISNULL(dbo.STUFFINGDETAIL.STFMID, 0) = 0)
AND (GIDATE >= '01-APR-2014')

--bASED ON sTUFFING
SELECT        dbo.STUFFINGDETAIL.STFMID, dbo.STUFFINGMASTER.STFMDNO, dbo.STUFFINGPRODUCTDETAIL.SBDID, dbo.SHIPPINGBILLDETAIL.GIDID, dbo.STUFFINGDETAIL.STFDID, dbo.SHIPPINGBILLDETAIL.SBMID, 
                         dbo.AUTHORIZATIONSLIPDETAIL.ASLDID, dbo.VEHICLETICKETDETAIL.VTDID, dbo.GATEINDETAIL.GIDID AS Expr1, dbo.GATEINDETAIL.CONTNRNO
FROM            dbo.STUFFINGDETAIL INNER JOIN
                         dbo.STUFFINGMASTER ON dbo.STUFFINGDETAIL.STFMID = dbo.STUFFINGMASTER.STFMID INNER JOIN
                         dbo.STUFFINGPRODUCTDETAIL ON dbo.STUFFINGDETAIL.STFDID = dbo.STUFFINGPRODUCTDETAIL.STFDID INNER JOIN
                         dbo.SHIPPINGBILLDETAIL ON dbo.STUFFINGPRODUCTDETAIL.SBDID = dbo.SHIPPINGBILLDETAIL.SBDID LEFT OUTER JOIN
                         dbo.GATEINDETAIL ON dbo.STUFFINGDETAIL.GIDID = dbo.GATEINDETAIL.GIDID LEFT OUTER JOIN
                         dbo.VEHICLETICKETDETAIL RIGHT OUTER JOIN
                         dbo.AUTHORIZATIONSLIPDETAIL ON dbo.VEHICLETICKETDETAIL.ASLDID = dbo.AUTHORIZATIONSLIPDETAIL.ASLDID ON dbo.STUFFINGDETAIL.STFDID = dbo.AUTHORIZATIONSLIPDETAIL.STFDID LEFT OUTER JOIN
                         dbo.VW_EXPORT_INVOICE_SBILL_NO_CBX_ASSGN_01 ON dbo.STUFFINGDETAIL.STFDID = dbo.VW_EXPORT_INVOICE_SBILL_NO_CBX_ASSGN_01.STFDID
WHERE        (ISNULL(dbo.VW_EXPORT_INVOICE_SBILL_NO_CBX_ASSGN_01.TRANDID, 0) = 0)
GROUP BY dbo.STUFFINGDETAIL.STFMID, dbo.STUFFINGMASTER.STFMDNO, dbo.STUFFINGPRODUCTDETAIL.SBDID, dbo.SHIPPINGBILLDETAIL.GIDID, dbo.STUFFINGDETAIL.STFDID, dbo.SHIPPINGBILLDETAIL.SBMID, 
                         dbo.AUTHORIZATIONSLIPDETAIL.ASLDID, dbo.VEHICLETICKETDETAIL.VTDID, dbo.GATEINDETAIL.GIDID, dbo.GATEINDETAIL.CONTNRNO
