CREATE PROCEDURE pr_BOE_Search_Container_No_Assgn @PBOENo varchar(100)
AS
BEGIN

	SET NOCOUNT ON;

	SELECT        TOP (100) PERCENT dbo.OPENSHEETDETAIL.GIDID, dbo.GATEINDETAIL.CONTNRNO
	FROM            dbo.OPENSHEETMASTER INNER JOIN
                         dbo.OPENSHEETDETAIL ON dbo.OPENSHEETMASTER.OSMID = dbo.OPENSHEETDETAIL.OSMID INNER JOIN
                         dbo.GATEINDETAIL ON dbo.OPENSHEETDETAIL.GIDID = dbo.GATEINDETAIL.GIDID
	WHERE        (dbo.OPENSHEETMASTER.BOENO = @PBOENo)
	ORDER BY dbo.GATEINDETAIL.CONTNRNO
END
GO
